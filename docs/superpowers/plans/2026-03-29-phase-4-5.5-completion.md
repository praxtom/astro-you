    const targetDate = date || new Date().toISOString().split('T')[0];
    const res = await fetch(`${API_BASE}/horoscope/personal/monthly`, {

            target_date: targetDate,
            horoscope_type: "monthly",
            language: "en", tradition: "universal",
            include_transits: true, include_progressions: false,
        }),
    });
    if (!res.ok) { console.warn("Monthly horoscope error:", res.status); return null; }
    return unwrap(await res.json());
}

const REMEDY_ROUTINE_MAP: Record<string, { type: 'morning' | 'evening' | 'meditation' | 'exercise'; durationMinutes: number }> = {
    Mantra:   { type: 'meditation', durationMinutes: 10 },

function canBeRoutine(category: string): boolean {
    return Object.keys(REMEDY_ROUTINE_MAP).some(k => category.toLowerCase().includes(k.toLowerCase()));
}

function getRoutineConfig(category: string) {
    const key = Object.keys(REMEDY_ROUTINE_MAP).find(k => category.toLowerCase().includes(k.toLowerCase()));
    return key ? REMEDY_ROUTINE_MAP[key] : null;
}

function addHeader(pdf: jsPDF, title: string, subtitle?: string) {
    pdf.setFillColor(3, 3, 8);
    pdf.rect(0, 0, 210, 297, 'F');

    pdf.setTextColor(212, 168, 83); // gold
    pdf.setFontSize(10);
    pdf.text('ASTROYOU', 20, 20);

    pdf.setTextColor(255, 255, 255);
    pdf.setFontSize(24);
    pdf.text(title, 20, 40);

    if (subtitle) {
        pdf.setTextColor(255, 255, 255, 0.5);
        pdf.setFontSize(10);
        pdf.text(subtitle, 20, 50);
    }

    // Separator line
    pdf.setDrawColor(212, 168, 83, 0.3);
    pdf.line(20, 55, 190, 55);
}

function addSection(pdf: jsPDF, title: string, y: number): number {
    pdf.setTextColor(212, 168, 83);
    pdf.setFontSize(11);
    pdf.text(title.toUpperCase(), 20, y);
    pdf.setDrawColor(255, 255, 255, 0.1);
    pdf.line(20, y + 2, 190, y + 2);
    return y + 10;
}

function addText(pdf: jsPDF, text: string, y: number, options?: { maxWidth?: number; color?: string; size?: number }): number {
    const maxWidth = options?.maxWidth || 170;
    const size = options?.size || 10;

    if (options?.color === 'muted') {
        pdf.setTextColor(255, 255, 255, 0.5);
    } else {
        pdf.setTextColor(255, 255, 255);
    }
    pdf.setFontSize(size);

    const lines = pdf.splitTextToSize(text, maxWidth);
    pdf.text(lines, 20, y);
    return y + lines.length * (size * 0.5) + 4;
}

function checkPageBreak(pdf: jsPDF, y: number, needed: number = 40): number {
    if (y + needed > 280) {
        pdf.addPage();
        pdf.setFillColor(3, 3, 8);
        pdf.rect(0, 0, 210, 297, 'F');
        return 20;
    }
    return y;
}

/**
 * Generate a Daily Forecast PDF
 */
export async function generateForecastPDF(
    forecast: any,
    period: string,
    userName?: string,
): Promise<void> {
    const pdf = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a4' });

    const dateStr = new Date().toLocaleDateString('en-US', {
        weekday: 'long', year: 'numeric', month: 'long', day: 'numeric'
    });

    addHeader(pdf, `${period.charAt(0).toUpperCase() + period.slice(1)} Forecast`, `${userName ? userName + ' — ' : ''}${dateStr}`);

    let y = 65;
    const h = forecast?.horoscope;

    // Theme
    if (forecast?.narrative || h?.overall_theme) {
        y = addSection(pdf, "Theme", y);
        y = addText(pdf, forecast.narrative || h.overall_theme, y);
    }

    // Life Areas
    if (h?.life_areas?.length) {
        y = checkPageBreak(pdf, y, 60);
        y = addSection(pdf, "Life Areas", y);
        for (const area of h.life_areas) {
            y = checkPageBreak(pdf, y, 20);
            pdf.setTextColor(255, 255, 255);
            pdf.setFontSize(10);
            pdf.text(`${area.area} (${area.rating}/5)`, 20, y);
            y += 5;
            y = addText(pdf, area.prediction || area.title, y, { color: 'muted', size: 9 });
        }
    }

    // Transit Activations
    if (h?.planetary_influences?.length) {
        y = checkPageBreak(pdf, y, 40);
        y = addSection(pdf, "Transit Activations", y);
        for (const inf of h.planetary_influences.slice(0, 6)) {
            y = checkPageBreak(pdf, y, 15);
            pdf.setTextColor(255, 255, 255);
            pdf.setFontSize(9);
            pdf.text(`${inf.planet} → ${inf.natal_planet} (${inf.strength})`, 20, y);
            y += 4;
            if (inf.description) {
                y = addText(pdf, inf.description, y, { color: 'muted', size: 8 });
            }
        }
    }

    // Lucky Elements
    if (h?.lucky_elements) {
        y = checkPageBreak(pdf, y, 30);
        y = addSection(pdf, "Lucky Elements", y);
        const le = h.lucky_elements;
        if (le.colors?.length) { y = addText(pdf, `Colors: ${le.colors.join(', ')}`, y, { size: 9 }); }
        if (le.numbers?.length) { y = addText(pdf, `Numbers: ${le.numbers.join(', ')}`, y, { size: 9 }); }
        if (le.stones?.length) { y = addText(pdf, `Stones: ${le.stones.join(', ')}`, y, { size: 9 }); }
        if (le.directions?.length) { y = addText(pdf, `Directions: ${le.directions.join(', ')}`, y, { size: 9 }); }
    }

    // Dasha
    if (forecast?.dasha?.mahadasha) {
        y = checkPageBreak(pdf, y, 20);
        y = addSection(pdf, "Active Dasha", y);
        y = addText(pdf, `Mahadasha: ${forecast.dasha.mahadasha}  |  Antardasha: ${forecast.dasha.bhukti || 'N/A'}`, y, { size: 9 });
    }

    // Footer
    pdf.setTextColor(255, 255, 255, 0.2);
    pdf.setFontSize(7);
    pdf.text('Generated by AstroYou — astroyou.app', 20, 290);

    pdf.save(`astroyou-${period}-forecast-${new Date().toISOString().split('T')[0]}.pdf`);
}

/**
 * Generate a Natal Chart PDF (with chart image capture)
 */
export async function generateNatalPDF(
    kundaliData: any,
    birthData: any,
    chartElementId: string,
    dashaInfo?: any,
    yogas?: any[],
    remedies?: any[],
): Promise<void> {
    const pdf = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a4' });

    addHeader(pdf, 'Natal Chart Report', `${birthData?.name || 'Jataka'} — Born ${birthData?.dob} at ${birthData?.tob}, ${birthData?.pob}`);

    let y = 65;

    // Capture chart image
    try {
        const chartBlob = await captureChartAsPng(chartElementId);
        const chartUrl = URL.createObjectURL(chartBlob);
        const img = new Image();
        img.src = chartUrl;
        await new Promise((resolve) => { img.onload = resolve; });

        const canvas = document.createElement('canvas');
        canvas.width = img.width;
        canvas.height = img.height;
        canvas.getContext('2d')?.drawImage(img, 0, 0);
        const imgData = canvas.toDataURL('image/png');

        const imgWidth = 120;
        const imgHeight = (img.height / img.width) * imgWidth;
        pdf.addImage(imgData, 'PNG', 45, y, imgWidth, imgHeight);
        y += imgHeight + 10;
        URL.revokeObjectURL(chartUrl);
    } catch {
        y = addText(pdf, 'Chart visualization unavailable', y, { color: 'muted' });
    }

    // Planetary Positions
    if (kundaliData?.planetary_positions?.length) {
        y = checkPageBreak(pdf, y, 60);
        y = addSection(pdf, "Planetary Positions", y);
        for (const p of kundaliData.planetary_positions) {
            y = checkPageBreak(pdf, y, 8);
            pdf.setTextColor(255, 255, 255);
            pdf.setFontSize(9);
            pdf.text(`${p.name}: ${p.sign} (House ${p.house})${p.is_retrograde ? ' [R]' : ''}${p.nakshatra ? ` — ${p.nakshatra}` : ''}`, 20, y);
            y += 5;
        }
    }

    // Dasha
    if (dashaInfo?.mahadasha) {
        y = checkPageBreak(pdf, y, 20);
        y = addSection(pdf, "Current Dasha Period", y);
        y = addText(pdf, `Mahadasha: ${dashaInfo.mahadasha}  |  Antardasha: ${dashaInfo.antardasha || 'N/A'}`, y, { size: 9 });
    }

    // Yogas
    if (yogas?.length) {
        y = checkPageBreak(pdf, y, 40);
        y = addSection(pdf, "Birth Yogas", y);
        for (const yoga of yogas.slice(0, 5)) {
            y = checkPageBreak(pdf, y, 10);
            pdf.setTextColor(255, 255, 255);
            pdf.setFontSize(9);
            pdf.text(`${yoga.name}${yoga.strength ? ` (${yoga.strength})` : ''}`, 20, y);
            y += 5;
        }
    }

    // Remedies
    if (remedies?.length) {
        y = checkPageBreak(pdf, y, 40);
        y = addSection(pdf, "Recommended Remedies", y);
        for (const r of remedies.slice(0, 5)) {
            y = checkPageBreak(pdf, y, 10);
            pdf.setTextColor(255, 255, 255);
            pdf.setFontSize(9);
            pdf.text(`${r.category}: ${r.name}`, 20, y);
            y += 4;
            if (r.detail) { y = addText(pdf, r.detail, y, { color: 'muted', size: 8 }); }
        }
    }

    // Footer
    pdf.setTextColor(255, 255, 255, 0.2);
    pdf.setFontSize(7);
    pdf.text('Generated by AstroYou — astroyou.app', 20, 290);

    pdf.save(`astroyou-natal-report-${new Date().toISOString().split('T')[0]}.pdf`);
}

/**

export async function generateCompatibilityPDF(
    compatibilityData: any,
    person1Name: string,
    person2Name: string,
    gunaData?: any,
): Promise<void> {
    const pdf = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a4' });

    addHeader(pdf, 'Compatibility Report', `${person1Name} & ${person2Name}`);

    let y = 65;

    if (compatibilityData?.overall_score) {

        pdf.setFontSize(18);
        pdf.text(`${compatibilityData.overall_score}%`, 20, y);
        y += 12;

        y = addText(pdf, `Total Score: ${gunaData.total_score}/36`, y);

                pdf.setTextColor(255, 255, 255);
                pdf.setFontSize(9);
                pdf.text(`${cat.name}: ${cat.obtained}/${cat.maximum}`, 20, y);
                y += 5;
            }

    // Composite
    if (compatibilityData?.composite) {

        }
    }

    // AI Narrative
    if (compatibilityData?.narrative) {
        y = checkPageBreak(pdf, y, 40);
        y = addSection(pdf, "AI Analysis", y);
        y = addText(pdf, compatibilityData.narrative, y, { size: 9 });
    }

    // Footer
    pdf.setTextColor(255, 255, 255, 0.2);
    pdf.setFontSize(7);
    pdf.text('Generated by AstroYou — astroyou.app', 20, 290);

    pdf.save(`astroyou-compatibility-${person1Name.toLowerCase()}-${person2Name.toLowerCase()}.pdf`);
}
